256  template<typename T>
257  inline void LuaWrapper::CreateUserdataContainer(lua_State* L, T* ptr)
258  {
259  // The userdata container which contains the lightuserdata (a T*)
260  typedef struct { T* ptr; } UserdataContainer;
261 
262  // Make sure the stack is big enough
263  lua_checkstack(L, 3);
264 
265  // Make a new userdata with the size of UserdataContainer
266  UserdataContainer* udata = (UserdataContainer*)lua_newuserdata(L, sizeof(UserdataContainer*));
267 
268  // Push the light userdata
269  lua_pushlightuserdata(L, ptr);
270 
271  // Copy the value at -2
272  lua_pushvalue(L, -2);
273 
274  // Set the lightuserdata (the pointer) to the newly created user data
275  lua_settable(L, -4);
276 
277  // Set the userdata's pointer to the pointer class's instance
278  udata->ptr = ptr;
279  }
